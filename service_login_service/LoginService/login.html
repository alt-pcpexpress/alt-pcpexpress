Future<bool> login(String email, String password) async {
  bool isLogin = false;
  try {
    final Response response = await ApiClient().get('login?username=$email&password=$password');

    final Map<String, dynamic> data = response.data;

    if (data['status'] == true && data['list'] != null && data['list'].isNotEmpty) {
      final userData = data['list'][0];
      final token = randomtoken;

      await UserInfo().setToken(token);
      await UserInfo().setUsername(userData['Username']);
      await UserInfo().setNama(userData['DisplayName']);
      await UserInfo().setUserID(userData['UserId'].toString());
      await UserInfo().setBranchId(userData['BranchId']);
      await UserInfo().setOficeSiteId(userData['OfficeSiteId']);
      await UserInfo().setCourierId(userData['CourierId'] ?? "CourierId Kosong");
      await UserInfo().setEmployeeId(userData['EmployeeId'] ?? "EmployeeId Kosong");
      await UserInfo().setUserImage(userData['UserImage'] ?? "UserImage/00000/7.png");

      isLogin = true;
    } else {
      await UserInfo().logout();
      isLogin = false;
    }
  } catch (e) {
    if (kDebugMode) {
      debugPrint('Error during login: $e');
    }
    await UserInfo().logout();
    isLogin = false;
  }

  return isLogin;
}
