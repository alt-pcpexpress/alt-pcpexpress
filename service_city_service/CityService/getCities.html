Future<List<City>> getCities(String districtName) async {
  try {
    final Response response = await ApiClient().get('mobile/CovrageArea?District_Name=$districtName&numberOfPage=100&pageNumber=0');
    final List<dynamic> dataList = response.data['list'];

    List<City> cities = dataList.map((cityData) => City.fromJson(cityData)).toList();
    return cities;
  } catch (e) {
    // Handle errors, e.g., log or throw custom exceptions
    if (kDebugMode) {
      debugPrint('Error fetching cities: $e');
    }
    return []; // Return an empty list on error
  }
}
