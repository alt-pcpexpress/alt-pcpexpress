Future<Map<String, dynamic>> saveDelivery({
  required Delivery delivery,
  required String groupstatus,
  required String status,
  required String remark,
  required String receiver,
  required List<Uint8List> imagedata,
  required Uint8List signatureData,
  required BuildContext context,
}) async {
  // Validation checks
  if (groupstatus.isEmpty) {
    return {'status': false, 'msg': 'Status di lokasi belum dipilih'};
  }
  if (status.isEmpty) {
    return {'status': false, 'msg': 'Status Pengantaran belum dipilih'};
  }
  if (remark.isEmpty) {
    remark = '';
  }
  if (receiver.isEmpty) {
    receiver = '';
  }
  if (groupstatus == "2") {
    if (imagedata.isEmpty) {
      return {'status': false, 'msg': 'Gambar kosong, silahkan lengkapi !'};
    }
    if (signatureData.isEmpty) {
      return {'status': false, 'msg': 'Tanda tangan wajib diisi'};
    }
    if (receiver.isEmpty) {
      return {'status': false, 'msg': 'Nama penerima wajib diisi !'};
    }
  }
  try {
    // Initialize the location plugin
    Location location = Location();
    UserInfo userInfo = UserInfo();

    // Check if service is enabled
    bool serviceEnabled = await location.serviceEnabled();
    if (!serviceEnabled) {
      serviceEnabled = await location.requestService();
      if (!serviceEnabled) {
        throw Exception('Location services are disabled.');
      }
    }

    // Check for location permission
    PermissionStatus permissionGranted = await location.hasPermission();
    if (permissionGranted == PermissionStatus.denied) {
      permissionGranted = await location.requestPermission();
      if (permissionGranted != PermissionStatus.granted) {
        throw Exception('Location permissions are denied');
      }
    }

    // Get the current location
    LocationData locationData = await location.getLocation();
    String courierid = await userInfo.getCourierId();
    String branchid = await userInfo.getBranchId();
    String officesiteid = await userInfo.getOfficeSiteId();
    String userid = await userInfo.getUserID();
    String seqNum = delivery.seqNumDsDtl.toString();
    String sekarang = DateFormat('yyyy-MM-dd HH:mm:ss').format(DateTime.now());
    if (seqNum.isEmpty || seqNum == "null") {
      seqNum = "0";
    }
    String url = "mobile/SaveDelivery";
    url += "?seqNum=$seqNum";
    url += "&dAId=${delivery.daIdDaHdr}";
    url += "&awbNo=${delivery.awbNoDaDtl}";
    url += "&dSDate=$sekarang";
    url += "&receiverName=$receiver";
    url += "&remark=$remark";
    url += "&groupStatusId=$groupstatus";
    url += "&statusId=$status";
    url += "&geotagLatitude=${locationData.latitude}";
    url += "&geotagLongitude=${locationData.longitude}";
    url += "&courierId=$courierid";
    url += "&branchId=$branchid";
    url += "&userId=$userid";
    url += "&officeSiteId=$officesiteid";

    List<Map<String, String>> photos = [];
    RandomString rs = RandomString();
    for (Uint8List image in imagedata) {
      String randomImageName = "${rs.getRandomString(
        uppersCount: 3,
        lowersCount: 3,
        numbersCount: 3,
        specialsCount: 1,
        specials: '_xx',
      )}.jpg";
      String base64Image = base64Encode(image);
      photos.add({'filename': randomImageName, 'data': base64Image});
    }

    String signame = '${DateTime.now().millisecondsSinceEpoch}.jpg';
    String base64Signature = base64Encode(signatureData);

    FormData formData = FormData.fromMap({
      for (int i = 0; i < photos.length; i++) ...{
        i == 0 ? 'pathFilePhoto' : 'pathFilePhoto$i': photos[i]['filename'],
        i == 0 ? 'dataFilePhoto' : 'dataFilePhoto$i': photos[i]['data'],
      },
      'pathFileSignature': signame,
      'dataFileSignature': base64Signature,
    });

    final Response response = await ApiClient().post(url, formData);
    if (response.statusCode == 200) {
      var jsonResponse = response.data;
      if (jsonResponse['status']) {
        return {'status': jsonResponse['status'], 'msg': 'Data berhasil disimpan'};
      } else {
        return {'status': jsonResponse['status'], 'msg': 'Data gagal disimpan (${jsonResponse['msg']})'};
      }
    } else {
      throw Exception('Failed to save data');
    }
  } catch (e) {
    return {'status': false, 'msg': 'Error: $e'};
  }
}
