Future<List<Province>> getProvince() async {
  try {
    final Response response = await ApiClient().get('Province');
    final List<dynamic> dataList = response.data['list'];

    List<Province> province = dataList
        .map((provinceData) => Province.fromJson(provinceData))
        .toList();
    return province;
  } catch (e) {
    // Handle errors, e.g., log or throw custom exceptions
    if (kDebugMode) {
      debugPrint('Error fetching cities: $e');
    }
    return []; // Return an empty list on error
  }
}
