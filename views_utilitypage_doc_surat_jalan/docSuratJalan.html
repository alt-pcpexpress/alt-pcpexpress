docSuratJalan(context, List<dynamic> sj) {
  return Container(
    //padding: const EdgeInsets.all(10),
    width: MediaQuery.of(context).size.width * 1,
    decoration: BoxDecoration(
      borderRadius: BorderRadius.circular(20),
      color: AppColors.hijaugelap,
    ),
    child: Column(
      children: [
        Container(
          alignment: Alignment.topCenter,
          padding: const EdgeInsets.only(bottom: 4, top: 4),
          decoration: const BoxDecoration(
            color: AppColors.hijaugelap,
            borderRadius: BorderRadius.only(
              topLeft: Radius.circular(20.0),
              topRight: Radius.circular(20.0),
            ),
          ),
          child: Padding(
            padding: const EdgeInsets.only(left: 14.0, right: 14.0),
            child: Row(
              mainAxisAlignment: MainAxisAlignment.spaceBetween,
              children: [
                const Text(
                  "Surat Jalan Kembali",
                  style: TextStyle(
                    color: AppColors.putih,
                    fontSize: 11,
                  ),
                  textAlign: TextAlign.center,
                ),
                Text(
                  "${sj.length} Dokumen",
                  style: const TextStyle(color: AppColors.putih, fontSize: 14, fontWeight: FontWeight.bold),
                  textAlign: TextAlign.center,
                ),
              ],
            ),
          ),
        ),
        Container(
          padding: const EdgeInsets.all(10),
          alignment: Alignment.centerLeft,
          height: 180,
          width: MediaQuery.of(context).size.width * 1,
          decoration: BoxDecoration(
            color: AppColors.putih,
            border: Border.all(color: Colors.grey),
            borderRadius: BorderRadius.circular(20),
          ),
          child: Column(
            mainAxisAlignment: MainAxisAlignment.start,
            crossAxisAlignment: CrossAxisAlignment.start,
            children: List.generate(sj.length, (index) {
              return Row(
                children: [
                  Text('${index + 1}. ${capitalizeFirstLetter(sj[index]['DocumentReference'].toString())}'),
                  IconButton(
                    icon: const FaIcon(
                      FontAwesomeIcons.fileImage,
                      size: 20,
                    ),
                    onPressed: () {
                      showImageDialog(context, sj[index]['image'].toString());
                    },
                  ),
                ],
              );
            }),
          ),
        ),
      ],
    ),
  );
}
