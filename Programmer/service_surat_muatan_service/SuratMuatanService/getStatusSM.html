Future<List<StatusSM>> getStatusSM() async {
  try {
    final Response response = await ApiClient().get('mobile/GetStatusSuratMuatan');
    final List<dynamic> dataList = response.data['list'];

    List<StatusSM> statussm = dataList.map((statusSM) => StatusSM.fromJson(statusSM)).toList();
    return statussm;
  } catch (e) {
    debugPrint('Error fetching Status SM: $e');
    return []; // Return an empty list on error
  }
}
