Future<List<Delivery>> delivery(String courierid) async {
  try {
    final Response response = await ApiClient().get(
      'mobile/DeliveryRecord?Courier_Id=$courierid',
    );

    if (response.data['status'] == false) {
      debugPrint("Delivery API response status is false.");
      return List.empty();
    } else {
      final List<dynamic> data = response.data['list'];
      List<Delivery> deliverys =
          data.map((service) => Delivery.fromJson(service)).toList();
      return deliverys;
    }
  } catch (error) {
    debugPrint("Error fetching Delivery data: $error");
    return List.empty();
  }
}
