Future<List<District>> getDistrict(String cityid) async {
  try {
    final Response response =
        await ApiClient().get('District?City_Id=$cityid');

    final List<dynamic> dataList = response.data['list'];
    List<District> district = dataList.map((district) {
      return District(
          districtId: district['District_Id'],
          cityId: district['City_Id'],
          cityName: district['City_Name'],
          districtName: district['District_Name']);
    }).toList();
    return district;
  } catch (e) {
    // Handle errors, e.g., log or throw custom exceptions
    if (kDebugMode) {
      debugPrint('Error fetching cities: $e');
    }
    return []; // Return an empty list on error
  }
}
