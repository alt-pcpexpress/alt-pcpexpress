Future<Map<String, dynamic>> addInvestigasi({
  required String awbNo,
  required String investigNotes,
  required DateTime investigDate,
}) async {
  try {
    UserInfo userInfo = UserInfo();
    var username = await userInfo.getUsername();
    var branchid = await userInfo.getBranchId();
    var officeid = await userInfo.getOfficeSiteId();
    var courierid = await userInfo.getCourierId();
    var officerid = await userInfo.getEmployeeId();
    var userid = await userInfo.getUserID();

    Map<String, dynamic> body = {
      'awbNo': awbNo,
      'branchId': branchid,
      'office_id': officeid,
      'courierId': courierid,
      'user_name': username,
      'statusId1': 203,
      'userId': userid,
      'investigNotes': investigNotes,
      'investig_officer_id': officerid,
      'investig_date': investigDate.toIso8601String(),
    };
    final Response response = await ApiClient().post("mobile/StoreInvestigasi", body);

    if (response.statusCode == 200) {
      return response.data as Map<String, dynamic>;
    } else {
      throw Exception('Failed to store investigation data');
    }
  } on DioException catch (e) {
    // Handle Dio-specific exceptions
    throw Exception('Failed to store investigation data: ${e.message}');
  }
}
