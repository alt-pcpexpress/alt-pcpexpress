Widget buildTunaiCard(String total, TextEditingController biayaController) {
  return FadeInUp(
    duration: const Duration(milliseconds: 300),
    child: Card(
      color: AppColors.hijau,
      child: Container(
        padding: const EdgeInsets.only(left: 10, right: 10, bottom: 10),
        child: Column(
          children: [
            Text(
              "Total Uang Tunai yang harus dibayarkan (Rp. ${total.replaceAllMapped(RegExp(r'(\d{1,3})(?=(\d{3})+(?!\d))'), (Match m) => '${m[1]}.')})",
              // ${NumberFormat("###.###", "id_ID").format(totalInt)})",
              textAlign: TextAlign.center,
              style: const TextStyle(fontSize: 10, color: AppColors.putih),
            ),
            Container(
              padding: const EdgeInsets.only(left: 20, right: 20),
              decoration: BoxDecoration(color: AppColors.putih, borderRadius: BorderRadius.circular(30)),
              child: TextFormField(
                controller: biayaController,
                decoration: const InputDecoration(prefixText: "Rp. ", prefixStyle: TextStyle(fontWeight: FontWeight.bold)),
                style: const TextStyle(
                  fontWeight: FontWeight.bold,
                  color: AppColors.dark,
                ),
                keyboardType: const TextInputType.numberWithOptions(decimal: false),
                inputFormatters: [
                  FilteringTextInputFormatter.digitsOnly,
                  CurrencyInputFormatter(),
                ],
              ),
            ),
          ],
        ),
      ),
    ),
  );
}
