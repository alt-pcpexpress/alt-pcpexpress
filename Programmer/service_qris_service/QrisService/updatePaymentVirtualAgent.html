Future<void> updatePaymentVirtualAgent({
  required String hpNo,
  required String awbNo,
  required String walletType,
  required double amountValue,
  double? adminFee,
}) async {
  final data = {
    'hp_no': hpNo,
    'awb_no': awbNo,
    'wallet_type': walletType,
    'amount_value': amountValue,
    'admin_fee': adminFee ?? 0,
  };

  try {
    final response = await ApiClient().post('apionline/UpdatePaymentVirtualAgentAsync', data);
    if (response.statusCode == 200) {
      // Handle the response if needed
      debugPrint('Payment updated successfully: ${response.data}');
    } else {
      // Handle error response
      debugPrint('Failed to update payment: ${response.data}');
    }
  } catch (e) {
    debugPrint('Error updating payment: $e');
    throw Exception('Error updating payment');
  }
}
